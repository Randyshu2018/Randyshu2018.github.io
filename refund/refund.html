# 退款相关API

# 创建退款申请 post /api/invoke

# 参数

参数名 类型 是否必填 说明
platformNo string 是 平台编号
orgName string 是 组织名称
enrollmentID string 是 用户名
password string 是 密码
describe.channel string 是 通道名称
describe.chaincode string 是 合约名称
describe.method string 是 合约方法(退款申请)
describe.ebaasUserId string 是 ebaas会员号
data.itemId string 是 订单条目编号
data.amount number 是 退款金额
data.refundNo string 是 退款编号(服务器接口获取)
data.refundAmountType string 是 退款金额类型(all(全额退款)/partial(部分退款))
data.receiptStatus string 是 货物状态(unreceipted(未收到商品)/receipted(已收到商品))
data.reason string 否 退款原因
data.instruction string 否 退款说明
data.evidence string 否 退款凭证

# 使用示例

请求参数:

{
  "platformNo": "E008",
  "orgName": "SampleOrg",
  "enrollmentID": "local-randy",
  "password": "123456",
  "describe": {
    "channel": "ch1",
    "chaincode": "commercial_platform",
    "method": "Refund:request",
    "ebaasUserId": "randy"
  },
  "data": {
    "itemId": "1_ORDER100",
    "amount": 20,
    "refundNo": "REFUND_1",
    "refundAmountType": "all",
    "receiptStatus": "receipted",
    "reason": "",
    "instruction": "",
    "evidence": ""
  }
}

# 确认退款 post /api/invoke

# 参数

参数名 类型 是否必填 说明
platformNo string 是 平台编号
orgName string 是 组织名称
enrollmentID string 是 用户名
password string 是 密码
describe.channel string 是 通道名称
describe.chaincode string 是 合约名称
describe.method string 是 合约方法:拒绝退款申请

    "method": "Refund:confirm",

    "ebaasUserId": "randy"
  },
  "data": {
    "refundNo": "REFUND_3",
    "paidAmount": 10

}
